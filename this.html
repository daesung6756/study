//간접 실행
function indirect1(){
    console.log(`HTML 안에서 간접 실행 : `, this);
}
const indirect1obj = {name:'My indirect1'};
indirect1.apply(indirect1obj) // {name:'My indirect1'}
indirect1.call(indirect1obj) // {name:'My indirect1'}
