function circumference(r) {
  return parseFloat(r) * 2.0 * Math.PI;
}
 
console.log(circumference(4.567));
// 28.695307297889173
 
console.log(circumference('4.567abcdefgh'));
// 28.695307297889173
 
console.log(circumference('abcdefgh'));
// NaN

Boolean() 논리형 데이터 타입으로 변경
첫 번재 매개변수로서 전달한 값은 필요한 경우 불리언 값으로 변환된다. 값이 없거나 0, -0, null, false, NaN, undefined, 빈 문자열 ("")이라면 객체의 초기값은 false가 된다.

Boolean("false");   // true
Boolean("");        // false
Boolean(null);      // false
Boolean(undefined); // false
Boolean([]);        // true
Boolean({});        // true
