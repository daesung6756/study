/* URL을 넣은 페이지메만 적용 시킬 때 url() */
@document url( '특정페이지.html') {
    h1 {font-size:3rem};
}
/* 특정 접두사가 사작된 경로에만 적용 시킬 때 use url-prefix() */
@document url-prefix('www.특정도메인.com/특정접두사/') {
    h1 {font-size:3rem};
}
/* 특정 도메인만 적용 시킬 때 domain() */
@document domain( 'www.특정도메인.com') {
    h1 {font-size:3rem};
}
/* 정규 표현식을 사용하여 처리 가능하다 regexp() */
@document regexp( "https:.*" ) {
    h1 {font-size:3rem};
}
