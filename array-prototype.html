const numbers = [ 1, 2, 3, 4, 5, 6 ];
 
console.log('includes( sunny ) :', friends.includes('sunny')); //false
console.log('includes( sunny ) :', friends.includes('Sunny')); //true
console.log('includes( 수진 ) :', friends.includes('수진')); //true
console.log('includes( 5 ) :', friends.includes(5)); //false
console.log('includes( 철수, 2 ) :', friends.includes('철수', 2)); //false
console.log('includes( 영희. 1 ) :', friends.includes('영희', 1)); //true
console.log('includes( undefined ) :', friends.includes( undefined )); //false
 
console.log('includes( 1 ) :', numbers.includes( 1 )); //true
console.log('includes( 7 ) :', numbers.includes( 7 )); //false
console.log('includes( 6, 5) :', numbers.includes( 6, 5 )); //true
console.log('includes( null ) :', numbers.includes( null )); //false

Array.prototype.reduce( )

배열의 각 요소에 대해 주어진 기존 배열에 합쳐서 새 배열을 반환한다.

reduce 함수는 4가지의 매개변수에서 정해져있는 인수를 받는다. reduce( function ( accumulator, currentValue, currentIndex , array ){});

  • 누신기 ( accumulator ) : 콜백의 반환 값을 누적
  • 현재값 ( currentValue )
    • -
  • 현재 인덱스 (currentIndex )
  • reduce를 호출한 배열 ( array ) :
initialValue 값을 정의 하지 않으면, reduce()는 index 1부터 시작해 콜백 함수를 실행하고, 첫번째 index를 생략한다. initialValue을 정의하면 index 0에서 시작한다.
