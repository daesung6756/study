Gulp Task

Gulp를 설치하면 cmd에서 "gulp" 명령어를 사용 할 수 있다.
해당 명령어를 사용하면 gulpfile.js 파일을 load 하게 된다.
내부에 task에 대한 setting을 할 수 있다.
gulpfile.js 내부는 보통
1.모듈 정의
2.리소스 테스크 정의
3.실행할 테스크를 정의
순으로 정리 한다.

// Scss resource는 아래와 같이 정의 할 수 있다.
gulp.task('scss:build', async function () {
    var options = {
        postcss: [ autoprefixer ({overrideBrowserslist: ['last 2 versions']})]
    }; // 자동으로 프리픽서를 붙여주는 버전을 정의
    return gulp.src('src/assets/scss/**/*.scss') // Scss 경로를 설정 
    .pipe(plumber()) // gulp 환경은 오류가 나면 튕기게 되어있다. 오류가 나도 안튕기게 처리하는 모듈이다.
    .pipe(scss({ // scss를 어떻게 빌드 할지 여러 옵션들을 설정 한다.
        fiber:Fiber,
        outputStyle: 'compressed', //compressed , expanded
        indentType: 'space',
        indentWidth: 1,
        compiler: dartSsss,
    })).on('error', scss.logError)
    .pipe(postcss(options.postcss))//이 시점에 pure css에 프리픽서를 붙도록 정의 한다.
    .pipe(gulp.dest('build/assets/css')) // 해당 폴더에 빌드 처리 한다.
    .pipe(browserSync.reload({ stream : true })); // 리소스에 뭔가 변화가 있을때 브라우저를 자동 새로고침하게 처리
})
