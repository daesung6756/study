화살표 함수란?

실제로 화살표(=>)를 사용하며, 기존의 함수를 조금 더 간결하게 표현 할 수 있도록 처리한 함수이다.
화살표 함수는 일반 함수를 100% 대체 할 수 없다.
그래서 모든 함수를 화살표 함수로 만들면 안된다.
상황에 맞게 쓰도록 해야 한다.

// 일반적이 표현식
const example1 = function(){
    console.log('함수 표현식')
}
 
// 화살표 함수
const example2 = () => {
    console.log('화살표 함수');
}

// 1. function 키워드 생략한다.
const example1 = () => {}
 
// 2. 매개변수가 1개라면 괄호()를 생략한다.
const example2 = 매개변수 => {}
 
// 3. 표현식이 1개라면 중괄호와 return을 생략한다.
const example3 = 매개변수 => 매개변수 + 1
 
// 4. this 가 정적으로 묶인다.
const UI = {
    init : function (){
        console.log('init', this ); // init function
        console.log('init', this === UI ); // true 출력
 
        // 중첩 선언식 함수
        function multiply() {
            console.log('선언식', this ); //window object
            console.log('선언식', this === UI ); // false 출력
        };
 
        // 중첩 화살표 함수
        const sum = num => {
            console.log('화살표', this ); // init function
            console.log('화살표', this === UI ); // true 출력
        };
        multiply(); // 선언식 함수 호출
        sum(); // 화살표 함수 호출
    }
}
