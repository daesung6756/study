for..in (es1)

for..in 문은 객체 순환 방식이라 index 값만 리턴을 한다.
value를 얻기위해선 추가적인 작업이 필요하기 때문에 배열 반복문으로는 추천하지 않는다.

//for..in
const names = ['철수', '영희', '길동', '강동', '미나', '다미'];
 
for (let item in names){
    console.log(item)
    // 0, 1, 2, 3, 4, 5
}

forEach (es5)

forEach는 배열 순환 방식이고, index와 value 값이 둘다 리턴되지만,
await 사용이 불가하며 for..of를 사용하거나 promise를 대체하여 사용 해야한다.
또한 불필요한 반복을 중지하기 위해서는 추가작업이 필요하기 때문에 코드가 전체적으로 복잡해진다.
es6를 사용 중이라면 배열 반복문으로 추천하지 않는다.

//forEach
const names = ['철수', '영희', '길동', '강동', '미나', '다미'];
 
names.forEach(function (value, index) {
    console.log(value);
    // 철수, 영희, 길동, 강동, 미나, 다미
    console.log(index);
    // 0, 1, 2, 3, 4, 5
})

for..of (es6)

for..of는 앞서 말한 반복문들의 부족한 부분이 업그레이드 된 모델이라고 생각하면 된다.
value와 index 값이 둘다 리턴이 되고, switch에서 사용했던 break 기능을 사용 할 수 있어서
조건에 의해 반복문을 중지하기도 간편하다.

나의 생각 : 현직 프론트 엔드 개발자들은 Array methods map() or reduce()를 반복문 대체용으로 많이 사용한다고 한다.

//for..of
const names = ['철수', '영희', '길동', '강동', '미나', '다미'];
 
for (let [index, value] of names.entries()){
    console.log(index);
    // 0, 1, 2, 3
    console.log(value);
    // 철수, 영희, 길동, 강동
 
    if(index === 3){
        break; // 반복 중지.
    }
}
